Node-Level Access Control Pattern
// Check what nodes a role can access MATCH (r:Role {name: 'DATA_ANALYST'}) -[:CAN_ACCESS]->(cl:Classification) <-[:HAS_CLASSIFICATION]-(n) RETURN labels(n)[0] AS node_type, cl.name AS classification, count(n) AS accessible_nodes
Column-Level Security Pattern (Role-Filtered Projection)
// Get columns accessible to a role for a given table WITH 'HR_MANAGER' AS current_role MATCH (r:Role {name: current_role}) -[:CAN_ACCESS]->(cl:Classification) <-[:CLASSIFIED_AS]-(col:Column {table_name: 'EMPLOYEES'}) RETURN col.name, col.data_classification, col.encryption_required, col.data_owner
Tag Propagation Verification
// Verify Snowflake tags were propagated to Neo4j column nodes MATCH (t:Table)-[:HAS_COLUMN]->(c:Column) -[:CLASSIFIED_AS]->(cl:Classification) WHERE c.snowflake_tagged = true RETURN t.name, c.name, cl.name AS classification, c.encryption_required, c.data_owner ORDER BY t.name, cl.sensitivity_rank
Access Denied Simulation
// Simulate: Can PUBLIC_USER access customer records? WITH 'PUBLIC_USER' AS current_role, 'Restricted' AS requested_class MATCH (r:Role {name: current_role}) OPTIONAL MATCH (r)-[:CAN_ACCESS]->(cl:Classification {name: requested_class}) RETURN current_role AS role, requested_class AS classification, CASE WHEN cl IS NULL THEN 'ACCESS DENIED' ELSE 'ACCESS GRANTED' END AS decision
Policy → Column Lineage
// See which masking policies protect which columns MATCH (p:Policy)-[:MASKS]->(c:Column) -[:CLASSIFIED_AS]->(cl:Classification) RETURN p.name AS policy, p.type AS policy_type, c.table_name + '.' + c.name AS column, cl.name AS classification ORDER BY p.name
